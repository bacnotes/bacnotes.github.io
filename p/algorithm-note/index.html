function charCount(str) {

  for (const char of str) {
    char = char.toLowerCase()
    if (isAlphaNumeric(char)) {
      char = char.toLowerCase()
      obj[char] = ++obj[char] || 1
    }
    return obj
  }
}

function isAlphaNumeric(char) {
  var code = char.charCodeAt(0)
  if (
    !(code > 47 && code < 58) && // 0-9
    !(code > 64 && code < 91) && // A-Z
    !(code > 96 && code < 123) // a-z
  ) {
    return false
  }
  return true
}
